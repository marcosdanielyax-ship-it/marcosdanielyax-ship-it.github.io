#include <iostream>
using namespace std;

int main() {
    for (int i = 1; i <= 100; ++i) {
        bool imprimioTexto = false;
        
        if (i % 3 == 0 && i % 5 == 0) {
            cout << "fizzbuzz";
            imprimioTexto = true;
        } 
        else if (i % 3 == 0) {
            cout << "fizz";
            imprimioTexto = true;
        } 
        else if (i % 5 == 0) {
            cout << "buzz";
            imprimioTexto = true;
        } 
        
        if (!imprimioTexto) {
            cout << i;
        }
        
        cout << endl; 
    }
    
    return 0;
}
