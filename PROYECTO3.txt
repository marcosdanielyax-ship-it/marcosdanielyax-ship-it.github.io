#include <iostream>
#include <string>
#include <algorithm>
#include <cctype>
using namespace std;

bool sonAnagramas(string palabraA, string palabraB) {
    palabraA.erase(remove(palabraA.begin(), palabraA.end(), ' '), palabraA.end());
    palabraB.erase(remove(palabraB.begin(), palabraB.end(), ' '), palabraB.end());

    transform(palabraA.begin(), palabraA.end(), palabraA.begin(), ::tolower);
    transform(palabraB.begin(), palabraB.end(), palabraB.begin(), ::tolower);

    if (palabraA == palabraB) return false;
    if (palabraA.length() != palabraB.length()) return false;

    sort(palabraA.begin(), palabraA.end());
    sort(palabraB.begin(), palabraB.end());

    return palabraA == palabraB;
}

int main() {
    string palabraUno, palabraDos;

    cout << "Ingrese primera palabra o frase: ";
    getline(cin, palabraUno);

    cout << "Ingrese segunda palabra o frase: ";
    getline(cin, palabraDos);

    if (sonAnagramas(palabraUno, palabraDos))
        cout << "Ambas palabras son anagramas." << endl;
    else
        cout << "Ambas palabras no son anagramas." << endl;

    return 0;
}
